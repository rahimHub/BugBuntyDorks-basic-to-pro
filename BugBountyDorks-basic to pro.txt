google
inurl:/bug-bounty
inurl:/security "bounty" OR "reward"
inurl:/responsible-disclosure "bounty" OR "reward" OR "swag"
inurl:/security-policy "bounty"
"submit vulnerability report" "reward"
"vulnerability disclosure policy" "bounty"
"security vulnerability report" "reward"
"responsible disclosure" "hall of fame"
"white hat program" "reward"
"bug bounty program" site:.*.*
۲. یافتن فایل‌های security.txt (استاندارد IETF):
google
inurl:/.well-known/security.txt "contact"
inurl:/security.txt "bounty" OR "reward"
site:*.* security.txt "bounty"
ext:txt "security" "contact" "bounty"
